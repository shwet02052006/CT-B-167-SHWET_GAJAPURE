Second employee info (from external DTD XML)

.then(response => response.text()) .then(data => { const parser = new DOMParser(); const xml = parser.parseFromString(data, 'application/xml'); const emps = xml.getElementsByTagName('employee'); const out = document.getElementById('output'); if (emps.length < 2) { out.textContent = 'No second employee found.'; return; } const emp = emps[1]; const pick = tag => emp.getElementsByTagName(tag)[0]?.textContent || ''; out.innerHTML = `

ID: ${pick('id')}

Name: ${pick('name')}

Department: ${pick('department')}

Designation: ${pick('designation')}

Email: ${pick('email')}

Salary: ${pick('salary')}

`; }) .catch(() => { document.getElementById('output').textContent = 'Failed to load XML.'; }); });
